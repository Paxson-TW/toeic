            renderQuestions();
            
            // 啟動倒數計時器
            startTimer();
        }

        // 2. 計時器控制
        function startTimer() {
            updateTimerDisplay();
            
            timer = setInterval(() => {
                timeLeft--;
                updateTimerDisplay();

                // 倒數 1 分鐘時給予紅色閃爍警告
                if (timeLeft <= 60) {
                    timerDisplayContainer.classList.add("timer-warning");
                }

                // 時間到強制收卷
                if (timeLeft <= 0) {
                    clearInterval(timer);
                    alert("⏱️ 測驗時間已到！系統將自動強制收卷並計算分數。");
                    finishExam(true); // true 表示是時間到被強制交卷
                }
            }, 1000);
        }

        function updateTimerDisplay() {
            const minutes = Math.floor(timeLeft / 60);
            const seconds = timeLeft % 60;
            timeLeftDisplay.textContent = 
                `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
        }

        // 3. 渲染測驗題目
        function renderQuestions() {
            questionsContainer.innerHTML = "";
            questions.forEach((q, index) => {
                const card = document.createElement("div");
                card.className = "question-card";
                card.setAttribute("id", `q-card-${q.id}`);

                let optionsHTML = "";
                for (const [key, value] of Object.entries(q.options)) {
                    optionsHTML += `
                        <li class="option-item">
                            <input type="radio" id="q${q.id}-${key}" name="q${q.id}" value="${key}" onchange="recordAnswer(${q.id}, '${key}')">
                            <label for="q${q.id}-${key}" class="option-label">
                                <strong>(${key})</strong> ${value}
                            </label>
                        </li>
                    `;
                }

                card.innerHTML = `
                    <div class="question-header">
                        <span class="question-number">${index + 1}</span>
                        <div class="question-text">${q.question}</div>
                    </div>
                    <ul class="options-list">
                        ${optionsHTML}
                    </ul>
                `;
                questionsContainer.appendChild(card);
            });
        }

        // 記錄學生選擇的答案
        window.recordAnswer = function(questionId, optionKey) {
            studentAnswers[questionId] = optionKey;
        };

        // 4. 計算分數與結束測驗
        function finishExam(isTimeOut = false) {
            isExamActive = false;
            clearInterval(timer);
            examEndTime = new Date();

            // 計算分數
            score = 0;
            questions.forEach(q => {
                if (studentAnswers[q.id] === q.correct) {
                    score++;
                }
            });

            // 處理未作答的預設值
            questions.forEach(q => {
                if (!studentAnswers[q.id]) {
                    studentAnswers[q.id] = "未作答";
                }
            });

            // 切換至結果畫面
            quizSection.style.display = "none";
            resultSection.style.display = "block";

            // 渲染結果數據
            displayScore.textContent = score;

            // 回饋評語
            if (score >= 9) {
                resultFeedback.textContent = "👑 優秀卓越！您已具備多益高分實力！";
            } else if (score >= 7) {
                resultFeedback.textContent = "👍 表現優異！觀念十分清晰，繼續加油！";
            } else if (score >= 5) {
                resultFeedback.textContent = "💡 還不錯！稍加複習弱點題型能更進一步。";
            } else {
                resultFeedback.textContent = "📚 加油！多讀多做解析是進步的不二法門。";
            }

            // 作答時間計算
            const elapsedTotalSeconds = Math.floor((examEndTime - examStartTime) / 1000);
            const elapsedMin = Math.floor(elapsedTotalSeconds / 60);
            const elapsedSec = elapsedTotalSeconds % 60;
            displayElapsed.textContent = `${elapsedMin} 分 ${elapsedSec} 秒`;

            // 格式化今日日期
            const today = new Date();
            displayDate.textContent = today.toLocaleDateString('zh-TW', {
                year: 'numeric',
                month: 'long',
                day: 'numeric',
                hour: '2-digit',
                minute: '2-digit'
            });
        }

        // 5. 渲染詳細對答解析
        function renderReviewMarkup() {
            reviewQuestions.innerHTML = "";
            questions.forEach((q, index) => {
                const card = document.createElement("div");
                const userAns = studentAnswers[q.id];
                const isCorrect = userAns === q.correct;

                card.className = `question-card ${isCorrect ? 'correct' : 'incorrect'}`;

                let badgeHTML = isCorrect 
                    ? `<span class="badge badge-correct">✓ 答對</span>`
                    : `<span class="badge badge-incorrect">✗ 答錯</span>`;

                let optionsListHTML = "";
                for (const [key, value] of Object.entries(q.options)) {
                    let textClass = "";
                    if (key === q.correct) textClass = "text-success font-weight-bold";
                    else if (key === userAns && !isCorrect) textClass = "text-danger";

                    optionsListHTML += `
                        <li style="margin-bottom: 6px; padding: 4px 8px; border-radius: 4px; ${key === q.correct ? 'background-color: #d1fae5;' : ''}">
                            <strong class="${textClass}">(${key}) ${value}</strong>
                            ${key === q.correct ? ' 🡄 <b>正確答案</b>' : ''}
                            ${key === userAns && !isCorrect ? ' 🡄 您的選擇' : ''}
                        </li>
                    `;
                }

                card.innerHTML = `
                    <div style="text-align: right;">${badgeHTML}</div>
                    <div class="question-header" style="margin-bottom: 10px;">
                        <span class="question-number" style="${isCorrect ? 'background-color: var(--success);' : 'background-color: var(--danger);'}">${index + 1}</span>
                        <div class="question-text">${q.question}</div>
                    </div>
                    <ul style="list-style: none; margin-left: 40px; margin-bottom: 15px;">
                        ${optionsListHTML}
                    </ul>
                    <div class="user-answer-indicator">
                        您的答案：<span class="${isCorrect ? 'text-success' : 'text-danger'}" style="font-size: 1.1rem; font-weight: 700;">${userAns}</span>
                    </div>
                    <div class="explanation">
                        <strong>💡 解析：</strong>${q.explanation}
                    </div>
                `;
                reviewQuestions.appendChild(card);
            });
        }

        // 6. 生成並下載文字檔成績報告
        function generateAndDownloadReport() {
            const today = new Date();
            const dateStr = today.toLocaleString('zh-TW');
            
            const elapsedTotalSeconds = Math.floor((examEndTime - examStartTime) / 1000);
            const elapsedMin = Math.floor(elapsedTotalSeconds / 60);
            const elapsedSec = elapsedTotalSeconds % 60;
            const timeStr = `${elapsedMin}分${elapsedSec}秒`;

            // 構建報告文字內容
            let reportText = `==================================================\n`;
            reportText += `       多益 Part 5 模擬練習 - 個人成績報告單       \n`;
            reportText += `==================================================\n\n`;
            reportText += `【基本資訊】\n`;
            reportText += `● 學生姓名/學號： ${studentName}\n`;
            reportText += `● 測驗日期： ${dateStr}\n`;
            reportText += `● 限制時間： 10 分鐘 (600秒)\n`;
            reportText += `● 作答總時間： ${timeStr}\n\n`;
            reportText += `【成績總覽】\n`;
            reportText += `● 測驗總題數： 10 題\n`;
            reportText += `● 答對題數： ${score} 題\n`;
            reportText += `● 答錯題數： ${10 - score} 題\n`;
            reportText += `● 正確勝率： ${(score / 10 * 100).toFixed(1)}%\n\n`;
            reportText += `【作答明細與答案對照】\n`;
            reportText += `--------------------------------------------------\n`;
            
            questions.forEach((q, index) => {
                const userAns = studentAnswers[q.id] || "未作答";
                const isCorrect = userAns === q.correct;
                const status = isCorrect ? "✓ [答對]" : "✗ [答錯]";
                
                reportText += `第 ${index + 1} 題: ${q.question}\n`;
                reportText += `  (A) ${q.options.A}\n`;
                reportText += `  (B) ${q.options.B}\n`;
                reportText += `  (C) ${q.options.C}\n`;
                reportText += `  (D) ${q.options.D}\n`;
                reportText += `  您的作答: ${userAns}   |  正確答案: ${q.correct}   --> ${status}\n`;
                reportText += `  解析: ${q.explanation}\n`;
                reportText += `--------------------------------------------------\n`;
            });
            
            reportText += `\n感謝您使用本測驗系統，請針對錯誤題型多加練習，祝您多益奪得高分！\n`;

            // 建立檔案下載連結
            const blob = new Blob([reportText], { type: "text/plain;charset=utf-8" });
            const link = document.createElement("a");
            link.href = URL.createObjectURL(blob);
            link.download = `${studentName}_多益練習成績單.txt`;
            
            // 觸發點擊事件下載
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }

        // 7. 重設考卷系統，準備下一次考試

            // 清理定時器

            identityInput.value = "";

            timerDisplayContainer.classList.remove("timer-warning");

            reviewBtn.textContent = "🔍 檢視完整答案解析";

            // 回到身分輸入畫面
            resultSection.style.display = "none";

        }
